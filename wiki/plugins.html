Constructor

Plugin constructor

Plugin(const std::string& name, const std::string& internal_name, const std::string& version, const std::string& description, const std::string& repository_url, std::shared_ptr<ILogger> logger)

Constructs a new Plugin instance with the specified metadata and logger.

Parameters

  • name const std::string& Display name of the plugin
  • internal_name const std::string& Internal identifier used for calling functions (lowercase, no spaces)
  • version const std::string& Plugin version (e.g., "1.0.0")
  • description const std::string& Brief description of the plugin's functionality
  • repository_url const std::string& URL to the plugin's repository or homepage
  • logger std::shared_ptr<ILogger> Shared pointer to the logger instance

Example

class ExamplePlugin : public RenWeb::Plugin {
public:
    ExamplePlugin(std::shared_ptr<RenWeb::ILogger> logger)
        : RenWeb::Plugin(
            "ExamplePlugin",           // name
            "example",                  // internal_name
            "1.0.0",                    // version
            "Example plugin",           // description
            "https://github.com/...",   // repository_url
            logger)                     // logger
    {
        logger->info("[example_plugin] Initializing...");
        registerFunctions();
    }
};

functions attribute

std::map<std::string, std::function<json::value(const json::value&)>> functions

Map of function names to their handler implementations. Register your plugin's functions in this map.

Example

// In constructor or registerFunctions() method:
functions["square"] = [this](const json::value& req) -> json::value {
    const json::value param = req.as_array()[0];
    if (param.is_int64()) {
        return json::value(param.as_int64() * param.as_int64());
    }
    return json::value(nullptr);
};

functions["reverse_string"] = [this](const json::value& req) -> json::value {
    const json::value param = req.as_array()[0];
    const std::string input = this->processInput(param).as_string().c_str();
    std::string reversed(input.rbegin(), input.rend());
    return this->formatOutput(json::value(reversed));
};

processInput instance

json::value processInput(const json::value& input)

Recursively decodes base64-encoded strings in a JSON value. Handles objects, arrays, and primitives.

Parameters

  • input const json::value& JSON value to decode

Returns

json::value - Decoded JSON value with strings converted from base64

Example

// JavaScript calls: renweb.call("example", "reverse_string", { text: "Hello" })
// Input arrives as: { text: { __encoding_type__: "base64", __val__: [72,101,108,108,111] } }

functions["reverse_string"] = [this](const json::value& req) -> json::value {
    const json::value param = req.as_array()[0];
    // Decode the base64-encoded string
    const std::string input = this->processInput(param).as_string().c_str();
    // input now contains: "Hello"
    std::string reversed(input.rbegin(), input.rend());
    return this->formatOutput(json::value(reversed));
};

Complete Example

Complete Plugin Implementation

This example demonstrates a complete plugin with multiple functions, proper error handling, and input/output processing. If you are looking for official pre-built plugin binaries, you can download them here.

C++ Plugin Implementation

// Define BOOST_JSON_SOURCE and include src.hpp in ONE .cpp file to enable header-only mode
#define BOOST_JSON_SOURCE
#include <boost/json/src.hpp>

#include "./plugin.hpp"

#if defined(_WIN32) || defined(_WIN64)
    #define PLUGIN_EXPORT __declspec(dllexport)
#elif defined(__GNUC__) || defined(__clang__)
    #define PLUGIN_EXPORT __attribute__((visibility("default")))
#else
    #define PLUGIN_EXPORT
#endif

namespace json = boost::json;

class ExamplePlugin : public RenWeb::Plugin {
public:
    ExamplePlugin(std::shared_ptr<RenWeb::ILogger> logger)
        : RenWeb::Plugin(
            "ExamplePlugin", 
            "example", 
            "1.0.0", 
            "Example plugin with math and utility functions", 
            "https://github.com/example/example_plugin", 
            logger) 
    {
        logger->info("[example_plugin] Initializing plugin...");
        registerFunctions();
        logger->info("[example_plugin] Plugin initialized successfully!");
    }

    ~ExamplePlugin() override { }

private:
    void registerFunctions() {
        // Square a number
        functions["square"] = [this](const json::value& req) -> json::value {
            try {
                const json::value param = req.as_array()[0];
                if (param.is_int64()) {
                    return json::value(param.as_int64() * param.as_int64());
                } else if (param.is_uint64()) {
                    return json::value(param.as_uint64() * param.as_uint64());
                } else if (param.is_double()) {
                    return json::value(param.as_double() * param.as_double());
                } else {
                    throw std::runtime_error("Invalid parameter type. Expected number.");
                }
            } catch (const std::exception& e) {
                this->logger->error(e.what());
                return json::value(nullptr);
            }
        };

        // Calculate factorial using gamma function
        functions["factorial"] = [this](const json::value& req) -> json::value {
            try {
                const json::value param = req.as_array()[0];
                if (param.is_int64()) {
                    return json::value(std::tgamma(param.as_int64()));
                } else if (param.is_uint64()) {
                    return json::value(std::tgamma(param.as_uint64()));
                } else if (param.is_double()) {
                    return json::value(std::tgamma(param.as_double()));
                } else {
                    throw std::runtime_error("Invalid parameter type. Expected number.");
                }
            } catch (const std::exception& e) {
                this->logger->error(e.what());
                return json::value(nullptr);
            }
        };

        // Reverse a string
        functions["reverse_string"] = [this](const json::value& req) -> json::value {
            try {
                const json::value param = req.as_array()[0];
                // Decode base64-encoded input string
                const std::string input = this->processInput(param).as_string().c_str();
                // Reverse the string
                std::string reversed(input.rbegin(), input.rend());
                // Encode output string to base64
                return this->formatOutput(json::value(reversed));
            } catch (const std::exception& e) {
                this->logger->error(e.what());
                return json::value(nullptr);
            }
        };
    }
};

extern "C" PLUGIN_EXPORT RenWeb::Plugin* createPlugin(std::shared_ptr<RenWeb::ILogger> logger) {
    return new ExamplePlugin(logger);
}

JavaScript Usage

/* ================== Important Notes ==================
* NOTE: It's up to you to create a Typescript module for the plugins you make! 
* NOTE: plugins are named as BIND_plugin_[internal_name]_[function_name] in the JS engine.
*       see below for examples
 ======================================================= */

// Square a number
const square = await BIND_plugin_example_square(7);
await Log.debug(square); // 49

// Calculate factorial
const fact = await BIND_plugin_example_factorial(5);
await Log.debug(fact); // 120 (gamma of 5)

// Reverse a string (automatic encoding/decoding)
const reversed = await BIND_plugin_example_reverse_string(Utils.encode("Hello")); // strings need encoding
await Log.debug(reversed); // "olleH"

// Get list of all plugins
const plugins = await Plugins.getPluginsList();
const examplePlugin = plugins.find(p => p.internal_name === "example");
await Log.debug(examplePlugin.name, examplePlugin.version);
await Log.debug("Available functions:", examplePlugin.functions);

Template

This template uses Boost.JSON in header-only mode for easy distribution without library dependencies. Download the plugin.hpp parent class to your project directory, then copy and paste this template to get started.

C++ Plugin Implementation

// Define BOOST_JSON_SOURCE and include src.hpp in ONE .cpp file to enable header-only mode
#define BOOST_JSON_SOURCE
#include <boost/json/src.hpp>

#include "./plugin.hpp"

#if defined(_WIN32) || defined(_WIN64)
    #define PLUGIN_EXPORT __declspec(dllexport)
#elif defined(__GNUC__) || defined(__clang__)
    #define PLUGIN_EXPORT __attribute__((visibility("default")))
#else
    #define PLUGIN_EXPORT
#endif

namespace json = boost::json;

/* ================== Important Notes ==================
* - Plugin functions in the JS engine are named as such:
*      - BIND_plugin_[internal_name]_[function_name]
* - The logger will be passed using dependency injection into the constructor.
*      - The logger can log trace, debug, info, warn, error, and critical messages.  
* - Boost.JSON is used in header-only mode (no separate library linking required).
*      - Define BOOST_JSON_SOURCE and include  in ONE .cpp file
*      - See: https://github.com/boostorg/json for Boost.JSON documentation and source
* - Compilation commands (header-only, no library linking needed):
*      - Linux:     g++ -std=c++17 -fPIC -shared custom_plugin.cpp -o libcustom_plugin.so -I./include -I/path/to/boost/include
*      - Windows:   cl /std:c++20 /utf-8 /bigobj /LD /EHsc /O2 /MT /I.\include /I\path\to\boost custom_plugin.cpp /link /DLL /out:custom_plugin.dll
*      - MacOS:     clang++ -std=c++17 -fPIC -shared -undefined dynamic_lookup -mmacosx-version-min=10.15 custom_plugin.cpp -o libcustom_plugin.dylib -I./include -I/path/to/boost/include
 ======================================================= */

class CustomPlugin : public RenWeb::Plugin {
public:
    CustomPlugin(std::shared_ptr<RenWeb::ILogger> logger)
        : RenWeb::Plugin(
            "CustomPlugin", 
            "custom",
            "0.0.0", 
            "Description about your plugin", 
            "https://github.com/your-repo", 
            logger) 
    {
        logger->info("[custom_plugin] Initializing plugin...");
        registerFunctions();
        logger->info("[custom_plugin] Plugin initialized successfully!");
    }

    ~CustomPlugin() override { }

private:
    void registerFunctions() {
        // Square a number
        functions["square"] = [this](const json::value& req) -> json::value {
            try {
                const json::array params = req.as_array(); // req will ALWAYS be an array.
                const json::value param1 = this->processInput(params[0]); // use processInput to decode in case using strings. If not, it'll just return the value as-is.
                const json::value param2 = this->processInput(params[1]); // you can have as many params as you'd like

                const json::value result = /* do something with param1 and param2 */;

                return this->formatOutput(result); // use formatOutput to encode the result, especially if it's a string. If not, it'll just return the value as-is.
                    // NOTE: both processInput and formatOutput will recursively encode/decode any json objects, arrays, and values.
                    // NOTE: formatOutput can also format std::strings in addition boost::json types.
            } catch (const std::exception& e) {
                this->logger->error(e.what());
                return json::value(nullptr);
            }
        };
    }
};

// THIS IS CRITICAL!! Keep this exact function signature at the end of your plugin implementation file so the RenWeb engine can load your plugin!!
extern "C" PLUGIN_EXPORT RenWeb::Plugin* createPlugin(std::shared_ptr<RenWeb::ILogger> logger) {
    return new ExamplePlugin(logger);
}
